Program
#include<stdio.h>

int main ()


{
    
     int a=4 ;
    
     double const b=8 ;
    
     char c='A';
    
     float d= 10.5 ;
    
     printf("The vale of a is %d\n",a) ;
    
     printf("The vale of b is %lf\n",b) ;
    
     printf("The vale of c is %c\n",c) ;
    
     printf("The vale of d is %.2f\n ",d) ;
   
     return 0 ;

}




OUTPUT

The vale of a is 4
The vale of b is 8.000000
The vale of c is A
The vale of d is 10.50
